    module WKB::DB

    Overview

    This module offers convenience converters when working with DB::Serializable from crystal-db.

    Example:

    require "db"
require "my_db_driver"
require "wkb"
require "wkb/db"

class MyClass
  include DB::Serializable

  property name : String

  @[DB::Field(converter: WKB::DB::ObjectConverter)]
  property geometry : WKB::Object
end
